    def secret2mw(self, v: str):
        # logger.debug(v)
        res = []
        # logger.debug(v.split("&#"))
        for s in v.split("&#"):
            if s:
                # logger.debug('0' + s)
                res.append(self.font['0' + s])
        return "".join(res)

    def get_position_info(self):
